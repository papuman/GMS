### Critical Section Protection

```python
# CRITICAL SECTION - Protected by advisory lock
def try_acquire_token(self):
    try:
        # 1. Acquire exclusive lock
        self._acquire_lock()  # pg_advisory_lock(123456789)

        # 2. Read current state (SELECT ... FOR UPDATE)
        state = self._get_state_table()

        # 3. Calculate refill
        current_tokens = self._refill_tokens(state)

        # 4. Check availability
        if current_tokens >= 1.0:
            # 5. Consume token
            new_tokens = current_tokens - 1.0

            # 6. Update state (UPDATE ... SET ...)
            self._update_state(new_tokens, ...)

            # 7. Commit
            self.env.cr.commit()

            return True
        else:
            return False

    finally:
        # 8. Always release lock
        self._release_lock()  # pg_advisory_unlock(123456789)
```

### 3. Clock Skew / Time Travel

```python
# Protection against time anomalies
def _refill_tokens(self, state):
    now = datetime.utcnow()
    last_refill = state['last_refill']

    elapsed = (now - last_refill).total_seconds()

    # Handle negative elapsed (clock went backwards)
    if elapsed < 0:
        _logger.warning(f"Clock skew detected: {elapsed}s")
        elapsed = 0  # Don't refill if time went backwards

    # Handle huge elapsed (server was down)
    if elapsed > 3600:  # 1 hour
        _logger.warning(f"Large time gap: {elapsed}s")
        # Still cap at bucket capacity

    tokens_to_add = elapsed * self.REFILL_RATE
    return min(state['tokens'] + tokens_to_add, self.BUCKET_CAPACITY)
```

### 4. Database Connection Loss

```python
# Advisory locks auto-release on connection close
# This prevents deadlocks if worker crashes

try:
    self._acquire_lock()
    # ... work ...
except Exception as e:
    # Lock automatically released by PostgreSQL
    _logger.error(f"Rate limiter error: {e}")
    raise
finally:
    # Explicitly release for clean shutdown
    self._release_lock()
```
